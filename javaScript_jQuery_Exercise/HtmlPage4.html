Write a program that determines whether a word is a Palindrome (the same backward as it is forward – “radar” for example). Display the result to the user.

Palindrome check:

                                function palindrome() {
                                    outputWord = "";
                                    inputWord = document.getElementById("word").value;
                                    for (var i = inputWord.length; i >= 0; i--) {
                                        outputWord += inputWord.charAt(i);
                                    }
                                    if (inputWord === outputWord) {
                                        $("#result").html(  inputWord + 
                                                            " is a Palindrome.");
                                    }
                                    else {
                                        $("#result").html(  inputWord + 
                                                            " is not a Palindrome.");
                                    }
                                }
